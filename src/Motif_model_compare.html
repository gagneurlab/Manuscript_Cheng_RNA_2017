def model(x, y, n_motif=5, lr=0.02, epochs=5000, model='poisson'):  
    n_motif = 5

    X = tf.placeholder(tf.float32,[None,n_motif])
    Y = tf.placeholder(tf.float32,[None,1])
    W = tf.Variable(tf.ones([n_motif,1]))
    
    if model == 'poisson':
        y_ = tf.log(tf.matmul(X,W))
    else:
        y_ = tf.matmul(X,W)
    
    cost_function = tf.reduce_mean(tf.square(y_ - Y))
    
    # fiting
    learning_rate = lr
    training_epochs = epochs
    init = tf.global_variables_initializer()
    optimizer = tf.train.GradientDescentOptimizer(learning_rate).minimize(cost_function)
    
    cost_history = np.empty(shape=[1],dtype=float)
    with tf.Session() as sess:
        sess.run(init)
        for epoch in range(training_epochs):
            sess.run(optimizer,feed_dict={X:x,Y:y})
            cost_history = np.append(cost_history,sess.run(cost_function,feed_dict={X: x,Y: y}))
        W_hat = sess.run(W)
        #calculate mean square error 
        pred_y = sess.run(y_, feed_dict={X: x})
        mse = tf.reduce_mean(tf.square(pred_y - y))
        mse = sess.run(mse)
        print("MSE: %.4f" % mse) 
    return W_hat, cost_history, mse, pred_y

def vis(y, pred_y, cost_history):
    #plot cost
    plt.plot(range(len(cost_history)),cost_history)
    #plt.axis([0,training_epochs,0,1])
    plt.show()

    fig, ax = plt.subplots()
    ax.scatter(y, pred_y)
    ax.plot([y.min(), y.max()], [y.min(), y.max()], 'k--', lw=3)
    ax.set_xlabel('Measured')
    ax.set_ylabel('Predicted')
    plt.show()
